QUERY THOUGHTS WITH REACTIONS
query {
  thoughts {
    _id
    username
    thoughtText
    reactions {
      _id
      createdAt
      username
      reactionBody
    }
  }
}

GET SINGLE USERNAME
query getSingleUser($username: String!) {
  user(username: $username) {
    username
    friendCount
    thoughts {
      thoughtText
      createdAt
    }
    friends {
      username
    }
  }
}
